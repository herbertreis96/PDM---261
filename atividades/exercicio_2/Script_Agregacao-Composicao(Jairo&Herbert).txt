class Dependente {
  late String _nome;
  
  String get nome => _nome;

  Dependente(String nome) {
    this._nome = nome;
  }
  
  // Método para converter para Map (útil para JSON)
  Map<String, dynamic> toJson() {
    return {
      'nome': _nome,
    };
  }
}

class Funcionario {
  late String _nome;
  late List<Dependente> _dependentes;
  
  String get nome => _nome;
  List<Dependente> get dependentes => _dependentes;

  Funcionario(String nome, List<Dependente> dependentes) {
    this._nome = nome;
    this._dependentes = dependentes;
  }
  
  // Método para converter para Map (útil para JSON)
  Map<String, dynamic> toJson() {
    return {
      'nome': _nome,
      'dependentes': _dependentes.map((dep) => dep.toJson()).toList(),
    };
  }
}

class EquipeProjeto {
  late String _nomeProjeto;
  late List<Funcionario> _funcionarios;
  
  String get nomeProjeto => _nomeProjeto;
  List<Funcionario> get funcionarios => _funcionarios;

  EquipeProjeto(String nomeprojeto, List<Funcionario> funcionarios) {
    _nomeProjeto = nomeprojeto;
    _funcionarios = funcionarios;
  }
  
  // Método para converter para Map (formato JSON)
  Map<String, dynamic> toJson() {
    return {
      'nomeProjeto': _nomeProjeto,
      'funcionarios': _funcionarios.map((func) => func.toJson()).toList(),
    };
  }
  
  @override
  String toString() {
    return toJson().toString();
  }
}

void main() {
  // 1. Criar varios objetos Dependentes
  var dependente1 = Dependente("Joãozinho Silva");
  var dependente2 = Dependente("Maria Santos");
  var dependente3 = Dependente("Pedro Oliveira");
  var dependente4 = Dependente("Ana Costa");
  var dependente5 = Dependente("Lucas Ferreira");
  var dependente6 = Dependente("Cosmo");
  var dependente7 = Dependente("Damiao");
  var dependente8 = Dependente("Raimundo Nonato");
  
  // 2. Criar varios objetos Funcionario
  // 3. Associar os Dependentes criados aos respectivos funcionarios
  var funcionario1 = Funcionario("Carlos Silva", [dependente1, dependente2]);
  var funcionario2 = Funcionario("Ana Souza", [dependente3]);
  var funcionario3 = Funcionario("Roberto Santos", [dependente4, dependente5]);
  var funcionario4 = Funcionario("Fernanda Lima", []); // Sem dependentes
  var funcionario5 = Funcionario("Jairo Jackson", [dependente6, dependente7]); 
  var funcionario6 = Funcionario("Herbert", [dependente8]); 

  
  // 4. Criar uma lista de Funcionarios
  List<Funcionario> listaFuncionarios = [
    funcionario1,
    funcionario2,
    funcionario3,
    funcionario4,
    funcionario5,
    funcionario6
  ];
  
  // 5. Criar um objeto Equipe Projeto chamando o metodo
  //    construtor que da nome ao projeto e insere uma
  //    coleção de funcionario
  var equipeProjeto = EquipeProjeto("Sistema de Gestão Empresarial", listaFuncionarios);
  
  // 6. Printar no formato JSON o objeto Equipe Projeto
  print("=== Dados da Equipe do Projeto em JSON ===\n");
  print(equipeProjeto.toString());
  
  // Print formatado para melhor visualização
  print("\n=== Visualização Detalhada ===\n");
  print("Projeto: ${equipeProjeto.nomeProjeto}");
  print("Total de Funcionários: ${equipeProjeto.funcionarios.length}\n");
  
  for (var funcionario in equipeProjeto.funcionarios) {
    print("Funcionário: ${funcionario.nome}");
    print("  Dependentes: ${funcionario.dependentes.length}");
    for (var dependente in funcionario.dependentes) {
      print("    - ${dependente.nome}");
    }
    print("");
  }
}
